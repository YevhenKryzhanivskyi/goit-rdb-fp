USE pandemic;
SHOW TABLES IN pandemic;

CREATE TABLE IF NOT EXISTS countries (
    country_id INT AUTO_INCREMENT PRIMARY KEY,
    country_name VARCHAR(100) NOT NULL,
    code VARCHAR(10) NOT NULL UNIQUE
);


CREATE TABLE IF NOT EXISTS infections (
    infection_id INT AUTO_INCREMENT PRIMARY KEY,
    infection_name VARCHAR(100) NOT NULL UNIQUE
);

CREATE TABLE IF NOT EXISTS infectious_cases_normalized (
    case_id INT AUTO_INCREMENT PRIMARY KEY,
    country_id INT NOT NULL,
    year INT NOT NULL,
    infection_id INT NOT NULL,
    cases INT DEFAULT 0,
    FOREIGN KEY (country_id) REFERENCES countries(country_id)
        ON DELETE CASCADE ON UPDATE CASCADE,
    FOREIGN KEY (infection_id) REFERENCES infections(infection_id)
        ON DELETE CASCADE ON UPDATE CASCADE
);

SELECT 
    c.country_name AS Entity,
    c.code AS Code,
    AVG(COALESCE(NULLIF(ic.Number_rabies, ''), 0)) AS avg_rabies,
    MIN(COALESCE(NULLIF(ic.Number_rabies, ''), 0)) AS min_rabies,
    MAX(COALESCE(NULLIF(ic.Number_rabies, ''), 0)) AS max_rabies,
    SUM(COALESCE(NULLIF(ic.Number_rabies, ''), 0)) AS sum_rabies
FROM infectious_cases ic
JOIN countries c ON c.code = ic.code
WHERE ic.Number_rabies <> ''   
GROUP BY c.country_name, c.code
ORDER BY avg_rabies DESC
LIMIT 10;

SELECT 
    c.country_name AS Entity,
    c.code AS Code,
    AVG(ic.cases) AS avg_rabies,
    MIN(ic.cases) AS min_rabies,
    MAX(ic.cases) AS max_rabies,
    SUM(ic.cases) AS sum_rabies
FROM infectious_cases_normalized ic
JOIN countries c ON ic.country_id = c.country_id
JOIN infections i ON ic.infection_id = i.infection_id
WHERE i.infection_name = 'rabies'
GROUP BY c.country_name, c.code
ORDER BY avg_rabies DESC
LIMIT 10;

DELIMITER //

CREATE FUNCTION diff_years_from_year(input_year INT)
RETURNS INT
DETERMINISTIC
BEGIN
    DECLARE start_of_year DATE;
    DECLARE years_diff INT;
    SET start_of_year = STR_TO_DATE(CONCAT(input_year, '-01-01'), '%Y-%m-%d');
    SET years_diff = TIMESTAMPDIFF(YEAR, start_of_year, CURDATE());
    RETURN years_diff;
END//

DELIMITER ;

SELECT 
    ic.year,
    diff_years_from_year(ic.year) AS diff_years
FROM infectious_cases_normalized ic
LIMIT 20;
